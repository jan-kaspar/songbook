\ifx\pdfoutput\undefined
	% the basic color macros
	\def\rgb#1{\special{color push rgb #1}}
	\def\cmyk#1{\special{color push cmyk #1}}
	\def\nocolor{\special{color pop}}

	\def\SetTextColor#1{\pdfliteral{#1 k}}
	\def\SetRuleColor#1{\pdfliteral{#1 K}}
	\def\cmyk#1{\SetTextColor{#1}\SetRuleColor{#1}}
\fi
